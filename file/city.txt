    function mapcompare(id, name, time) {
        var cond = {};
        cond.INDEX_ID = id;
        cond.SHOW_NAME = name;
        cond.DATE_TIME = time;
        cond.CITY_GROUP = $(".selcity b").html();
        cond.COLOR_COUNT = 4;
        cond.COLOR_DETAIL = "0|5|#8fcbee;5|25|#62b4e8;25|50|#62a1e8;50|100|#5486d8";
        var url = "/City/CityMapTools/MapToolReirect" + "?Param=" + encodeURI(JSON.stringify(cond).replace(/\#/g, "jhjth"));
        window.open(url);
    }
